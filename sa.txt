//SWAPPING of 2 numbers using 3rd variable.

#include<stdio.h>
int main()
{
  int a=10,b=20,c;
  printf("Before swapping A=%d B=%d",a,b);
  c=a;
  a=b;
  b=c;
  print("After swapping a=%d B=%d",a,b);
}

//SWAPPING of 2 numbers without using third variable

#include<stdio.h>
int main()
{
  int a=10,b=20;
  printf("Before swapping A=%d B=%d",a,b);
  a=a+b;
  b=a-b;
  a=a-b;
  print("After swapping A=%d B=%d",a,b);
}
